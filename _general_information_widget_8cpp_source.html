186 bool
187 GeneralInformationWidget::outputToJSON(QJsonObject &jsonObj){
188 
189  jsonObj["name"] = nameEdit->text().trimmed();
190  jsonObj["revision"] = revEdit->text().toDouble();
191  jsonObj["type"] = typeEdit->text().trimmed();
192  jsonObj["year"] = yearBox->text().toInt();
193  jsonObj["stories"] = storiesBox->text().toInt();
194  jsonObj["width"] = widthEdit->text().toDouble();
195  jsonObj["depth"] = depthEdit->text().toDouble();
196 
197  // if the plan area is provided:
198  double planAreaValue = planAreaEdit->text().toDouble();
199  if (planAreaValue > 0.0)
200  jsonObj["planArea"] = planAreaValue;
201  else {
202  double widthValue = widthEdit->text().toDouble();
203  double depthValue = depthEdit->text().toDouble();
204  double rectAreaValue = widthValue * depthValue;
205  // otherwise, assume a rectangular plan and use width x depth
206  if (rectAreaValue > 0.0)
207  jsonObj["planArea"] = rectAreaValue;
208  else
209  // if none of the above are provided, use 0.0 for the area
210  jsonObj["planArea"] = 0.0;
211  }
212 
213  jsonObj["height"] = heightEdit->text().toDouble();
214  jsonObj["weight"] = weightEdit->text().toDouble();
215 
216  QJsonObject location;
217  location["name"] = locationNameEdit->text().trimmed();
218 
219  QString dblVal = locationLatBox->text();
220  location["latitude"] = dblVal.toDouble();
221 
222  dblVal = locationLonBox->text();
223  location["longitude"] = dblVal.toDouble();
224 
225  jsonObj["location"] = location;
226 
227  QJsonObject units;
228  units["force"] = unitEnumToString(unitsForceCombo->currentData().value<ForceUnit>());
229  units["length"] = unitEnumToString(unitsLengthCombo->currentData().value<LengthUnit>());
230  units["time"] = unitEnumToString(unitsTimeCombo->currentData().value<TimeUnit>());
231  units["temperature"] = unitEnumToString(unitsTemperatureCombo->currentData().value<TemperatureUnit>());
232 
233  jsonObj["units"] = units;
234 
235  return(true);
236 
237 }
